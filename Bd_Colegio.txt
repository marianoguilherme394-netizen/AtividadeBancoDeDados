create database Colegio
use Colégio

create table tbAlunos(
matricula varchar(5) primary key auto_increment,
nome varchar(50) NOT NULL,
endereco varchar(50) NOT NULL,
cidade varchar(30) NOT NULL,
codCurso);

create table tbCursos(
codAluno char(2) primary key auto_increment,
nome varchar(50) NOT NULL,
codDisc1 char(2), constraint Fk_codDisc1 foreign key(codDisciplina) references tbDisciplinas (codDisciplina),
codDisc2 char(2), constraint Fk_codDisc2 foreign key(codDisciplina) references tbDisciplinas (codDisciplina),
codDisc3 char(2), constraint Fk_codDisc3 foreign key(codDisciplina) references tbDisciplinas (codDisciplina));

create table tbDisciplinas(
codDisciplina char(2) primary key auto_increment,
nomeDisciplina varchar(30) NOT NULL);
